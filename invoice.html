// Ambil ID Invoice dari URL (misal: website.com/invoice?id=INV-002) const urlParams = new URLSearchParams(window.location.search); const invoiceId = urlParams.get('id'); // Cek status pembayaran setiap 3 detik const checkInterval = setInterval(async () => { try { const response = await fetch(`/api/invoice-status/${invoiceId}`); const data = await response.json(); if (data.status === 'PAID') { clearInterval(checkInterval); // Hentikan pengecekan // Beri notifikasi keren sebelum dilempar alert("Pembayaran Sukses! Anda akan dialihkan ke Sosial Media Admin..."); // Redirect otomatis ke link WhatsApp / Telegram yang diset admin window.location.href = data.redirectUrl; } } catch (error) { console.error("Gagal mengecek status pembayaran", error); } }, 3000); // 3000ms = 3 detik
